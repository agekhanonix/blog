# |>controller          Les controleurs de l'application
# |>divers              Un fourre-tout pour tout ce qui pourait être utilie
# |>libs                Avec entre-autres chose des classes PHP

# |>model               Les différents "model" utilisés
# |>public              On y retrouve le CSS, le JS, les IMAGES, le CMS BOOTSTRAP
#        |>bootstrap    Le CMS
#        |>css          le/les css de l'application
#        |>images       le/les images de l'application
#               |>ico
#               |>slider
#               |>articles
#        |>js           les scripts javascript du site 
#        |>tinymce      L'outil "tinymce"
#        |>objects      Les objets JS au sens POO
# |>template            Le template d'affichage du site
# |>view                Les # vues d'affichage
#      |>frontend       Les # vues du "front office"
